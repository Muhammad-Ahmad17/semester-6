\section{Problem 3.4: Bit Plane Slicing Transformations}
%══════════════════════════════════════════════════════════════════════════════

\begin{questionbox}
\textbf{Problem 3.4:} A common implementation of intensity slicing is to highlight one range of intensities $[A, B]$ by mapping it to white (255) and mapping all other intensities to black (0). Propose a set of intensity-slicing transformations capable of producing all the individual bit planes of an 8-bit monochrome image.
\end{questionbox}

\subsection{Understanding Bit Planes (Section 3.2.4)}

\begin{conceptbox}[What is a Bit Plane?]
An 8-bit image has pixel values 0-255. Each value can be written as 8 binary digits. A \textbf{bit plane} extracts just ONE bit position from every pixel, creating a binary image.

\textbf{Pixel decomposition:}
\[
\text{pixel value} = \sum_{k=0}^{7} b_k \cdot 2^k = b_7 \cdot 128 + b_6 \cdot 64 + b_5 \cdot 32 + \cdots + b_0 \cdot 1
\]
where $b_k \in \{0, 1\}$ is the $k$-th bit.
\end{conceptbox}

\[
167 = 128 + 32 + 4 + 2 + 1 = 1 \cdot 2^7 + 0 \cdot 2^6 + 1 \cdot 2^5 + 0 \cdot 2^4 + 0 \cdot 2^3 + 1 \cdot 2^2 + 1 \cdot 2^1 + 1 \cdot 2^0

\subsection{The Solution: Intensity Slicing Transformation}

\begin{keypoint}
To extract bit plane $k$, use intensity slicing that maps to 255 where bit $k$ is 1:

\textbf{Transformation function:}
\[
T_k(r) = \begin{cases}
255 & \text{if } \left\lfloor \dfrac{r}{2^k} \right\rfloor \mod 2 = 1\\[0.8em]
0 & \text{if } \left\lfloor \dfrac{r}{2^k} \right\rfloor \mod 2 = 0
\end{cases}
\]

\textbf{How it works:}
\begin{enumerate}
    \item $\lfloor r/2^k \rfloor$ = shift binary right by $k$ positions
    \item $\mod 2$ = extract the rightmost bit (which was bit $k$)
\end{enumerate}
\end{keypoint}

\section{Problem 3.5: Effect of Zeroing Bit Planes on Histogram}

\begin{questionbox}
\textbf{Problem 3.5:} Discuss the effect that setting to zero the lower-order bit planes would have on the histogram of an image in general. What would be the effect on the histogram if we set to zero the higher-order bit planes instead?
\end{questionbox}

\subsection{Part (a): Zeroing Lower-Order Bits}

\begin{conceptbox}[Setting Lower Bits to Zero]
Zeroing the $k$ least significant bits forces all values to be \textbf{multiples of $2^k$}:
\[
\text{new value} = \left\lfloor \frac{\text{old value}}{2^k} \right\rfloor \times 2^k
\]
\end{conceptbox}

\end{conceptbox}

    \item 255 = \textcolor{red}{1111}1111 $\rightarrow$ \textcolor{red}{0000}1111 = 15
\end{itemize}

\section{Problem 3.7: Second Pass Produces Same Result}
%══════════════════════════════════════════════════════════════════════════════

\begin{questionbox}
\textbf{Problem 3.7:} Prove that a second pass of histogram equalization will yield exactly the same result as the first pass.
\end{questionbox}

\subsection{Understanding the Question}

\begin{conceptbox}[What We Need to Prove]
If we apply histogram equalization to an image:
\begin{enumerate}
    \item \textbf{First pass:} Original image $f$ $\rightarrow$ Equalized image $g$
    \item \textbf{Second pass:} Equalized image $g$ $\rightarrow$ Result $h$
\end{enumerate}

\textbf{Claim:} $h = g$ (the second pass doesn't change anything!)
\end{conceptbox}

\subsection{Recap: Histogram Equalization Formula}

The transformation function for histogram equalization is:
\[
s = T(r) = (L-1) \int_0^r p_r(w) \, dw = (L-1) \cdot \text{CDF}(r)
\]

where:
\begin{itemize}
    \item $r$ = input intensity (0 to $L-1$)
    \item $s$ = output intensity
    \item $p_r(r)$ = probability density function (normalized histogram) of input
    \item CDF = cumulative distribution function
    \item $L$ = number of gray levels (typically 256)
\end{itemize}

\subsection{Step 1: Result of First Equalization}

\begin{conceptbox}[After First Pass]
After histogram equalization, the output $s$ has a \textbf{uniform distribution}:
\[
p_s(s) = \frac{1}{L-1} \quad \text{for } 0 \leq s \leq L-1
\]

This is the \textbf{goal} of histogram equalization --- to spread intensities evenly across the entire range.
\end{conceptbox}

\begin{center}
\begin{tikzpicture}[scale=0.6]
% Before equalization
\begin{scope}

\draw[->] (0,0) -- (0,4) node[above] {$p_r(r)$};

\subsection{Step 2: Apply Equalization Again (Second Pass)}

Now we apply histogram equalization to the already-equalized image $s$:
\[
z = T(s) = (L-1) \int_0^s p_s(w) \, dw
\]

Since $p_s(s) = \dfrac{1}{L-1}$ (uniform):
\begin{align*}
z &= (L-1) \int_0^s \frac{1}{L-1} \, dw \\[0.5em]
&= (L-1) \cdot \frac{1}{L-1} \int_0^s dw \\[0.5em]
&= (L-1) \cdot \frac{1}{L-1} \cdot s \\[0.5em]
&= s
\end{align*}

\begin{keypoint}
\textbf{Result:} $z = s$

The second equalization produces the \textbf{identity transformation} --- output equals input!
\end{keypoint}

\subsection{Why Does This Happen?}

\begin{conceptbox}[The Mathematical Reason]
The CDF of a uniform distribution is a \textbf{straight line}:
\[
\text{CDF}(s) = \int_0^s \frac{1}{L-1} dw = \frac{s}{L-1}
\]

When we multiply by $(L-1)$:
\[
T(s) = (L-1) \cdot \frac{s}{L-1} = s
\]

A straight-line CDF produces the \textbf{identity transformation}!
\end{conceptbox}

\subsection{Practical Implication}

\begin{keypoint}
\textbf{Conclusion:}
\begin{enumerate}
    \item Histogram equalization is \textbf{idempotent}: applying it multiple times gives the same result as applying it once
    \item Once an image is equalized, it stays equalized --- no further ``improvement'' is possible
    \item This is because a uniform distribution is already the ``optimal'' distribution that equalization tries to achieve
\end{enumerate}

\textbf{In practice:} If you accidentally run histogram equalization twice on an image, don't worry --- the result is the same as running it once!
\end{keypoint}

\subsection{The Answer and Condition}

\begin{keypoint}
\textbf{Answer for ALL operations $(+, -, \times, \div)$:}

You \textbf{CANNOT} compute the result histogram from $h_f$ and $h_g$ alone under general conditions.

\textbf{Proof by counterexample:}
\begin{itemize}
    \item $h_B = h_C$ (identical individual histograms)
    \item But $h_{A+B} \neq h_{A+C}$ (completely different result histograms)
\end{itemize}

\textbf{What information is needed instead:}

The \textbf{joint histogram} $h(f,g)$ which counts how many pixels have $f(x,y)=a$ AND $g(x,y)=b$ simultaneously.
\end{keypoint}

\subsection{Special Cases Where It IS Possible}

\begin{conceptbox}[When Can We Compute Result Histogram?]
Only under very restrictive conditions:

\textbf{1. Constant image:} If one image is constant ($g(x,y) = c$ for all pixels)
\begin{itemize}
    \item Then $h_{f+g}$ is just $h_f$ shifted by $c$
\end{itemize}

\textbf{2. Independent images:} If the locations of intensity values in $f$ and $g$ are statistically independent (random), we can compute expected histogram using probability.

\textbf{3. Same image:} If $g = f$, then $h_{f+f} = h_{2f}$ can be computed.

In general practical cases, \textbf{none of these hold}.
\end{conceptbox}

\begin{questionbox}
\textbf{Problem 3.15:} (a) Develop a procedure for computing the box-filter as the filter moves from point to point. (b) Determine the computational advantage over straightforward implementation.
\end{questionbox}

\subsection{Understanding Box Filter (Section 3.5.1)}

\begin{conceptbox}[What is a Box Filter?]
A \textbf{box filter} (also called \textit{averaging filter} or \textit{mean filter}) replaces each pixel with the \textbf{average} of all pixels in its $n \times n$ neighborhood.

\textbf{Mathematical Definition:}
\[
g(x,y) = \frac{1}{n^2} \sum_{i=-k}^{k} \sum_{j=-k}^{k} f(x+i, y+j) \quad \text{where } n = 2k+1
\]

\textbf{Applications:}
\begin{itemize}
    \item Noise reduction (smoothing)
    \item Blurring for privacy/artistic effects
    \item Preprocessing for edge detection
\end{itemize}
\end{conceptbox}

For each output pixel, we must sum ALL $n^2$ values in the window:

\subsection{Part (a): The Efficient Sliding Window Algorithm}

\begin{conceptbox}[Key Insight]
When the window slides one pixel right, \textbf{only ONE column changes}:
\begin{itemize}
    \item One column \textbf{leaves} on the left $\rightarrow$ \textbf{subtract} its sum
    \item One column \textbf{enters} on the right $\rightarrow$ \textbf{add} its sum
\end{itemize}

The other $(n-1)$ columns stay the same --- no need to recompute!
\end{conceptbox}

\subsection{Complete Algorithm Steps}

\begin{enumerate}
    \item \textbf{Initialize:} Compute full sum for first window position: $S_0 = \sum_{i,j} f(i,j)$

    \item \textbf{Precompute Column Sums:} Store the $n$-pixel sum of each column in the row
    \[
    C_j = \sum_{i=0}^{n-1} f(x+i, j) \quad \text{for each column } j
    \]

    \item \textbf{Slide Horizontally:} For each step right:
    \[
    S_{x+1} = S_x - C_{left} + C_{right}
    \]
    Average = $S/(n^2)$

    \item \textbf{Move to Next Row:} Update column sums (subtract top row, add bottom row):
    \[
    C_j^{new} = C_j^{old} - f(top, j) + f(bottom, j)
    \]
\end{enumerate}

\subsection{Part (b): Computational Advantage Analysis}

\begin{conceptbox}[Operation Count Comparison]
For an $n \times n$ filter on an $M \times N$ image:

\textbf{Naive Method:}
\begin{itemize}
    \item Operations per pixel: $n^2$ additions
    \item Total: $M \times N \times n^2$ operations
\end{itemize}

\textbf{Efficient Sliding Method:}
\begin{itemize}
    \item Horizontal slide: 2 operations (subtract left + add right)
    \item Update column sums per row: $n$ columns $\times$ 2 ops = $2n$ operations total
    \item Average per pixel: $\approx 2n$ operations
\end{itemize}
\end{conceptbox}

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{|l|c|c|c|c|}
\hline
\textbf{Method} & \textbf{Ops/pixel} & $n=5$ & $n=21$ & $n=101$ \\
\hline
Naive & $n^2$ & 25 & 441 & 10,201 \\
Efficient & $2n$ & 10 & 42 & 202 \\
\hline
\textbf{Speedup} & $\dfrac{n^2}{2n} = \dfrac{n}{2}$ & $\mathbf{2.5\times}$ & $\mathbf{10.5\times}$ & $\mathbf{50.5\times}$ \\
\hline
\end{tabular}
\end{center}

\begin{keypoint}
\textbf{Key Results:}
\begin{enumerate}
    \item Speedup factor = $\dfrac{n}{2}$ (larger filters benefit more!)
    \item Time complexity: $O(n^2) \rightarrow O(n)$ per pixel
    \item For large filters (e.g., $101 \times 101$), get \textbf{50× speedup}
    \item Algorithm is \textbf{independent of filter size} for horizontal traversal --- each step is just 2 operations!
\end{enumerate}
\end{keypoint}

\section{Problem 3.19: Computing Median Efficiently}
%══════════════════════════════════════════════════════════════════════════════

\begin{questionbox}
\textbf{Problem 3.19:} (a) Give a procedure for computing the median of an $n \times n$ neighborhood. (b) Propose a technique for updating the median as the center moves from pixel to pixel.
\end{questionbox}

\subsection{Understanding Median Filter (Section 3.5.2)}

\begin{conceptbox}[What is Median Filtering?]
The \textbf{median filter} replaces each pixel with the \textbf{median} (middle value when sorted) of all pixels in its neighborhood.

\textbf{Why use median instead of mean?}
\begin{itemize}
    \item \textbf{Removes salt-and-pepper noise} --- extreme outliers don't affect the median
    \item \textbf{Preserves edges} --- unlike mean filter which blurs edges
    \item The median is a \textbf{robust statistic} --- less sensitive to outliers
\end{itemize}
\end{conceptbox}

\subsection{Part (a): Basic Median Computation Procedure}

\begin{enumerate}
    \item \textbf{Extract:} Collect all $n^2$ pixel values from the $n\times n$ neighborhood
    \item \textbf{Sort:} Arrange values in ascending (or descending) order
    \item \textbf{Select:} Median = value at position $\dfrac{n^2+1}{2}$
\end{enumerate}

\subsection{Why Sorting is Expensive}

\begin{conceptbox}[Computational Cost of Basic Approach]
\textbf{For each pixel:}
\begin{itemize}
    \item Extract $n^2$ values: $O(n^2)$
    \item Sort using comparison sort: $O(n^2 \log n^2) = O(n^2 \log n)$
    \item Select median: $O(1)$
\end{itemize}

\textbf{Total per pixel:} $O(n^2 \log n)$

For a $1000 \times 1000$ image with $21 \times 21$ filter:
\[
1000 \times 1000 \times 441 \times \log(441) \approx 3.8 \times 10^{12} \text{ comparisons!}
\]
\end{conceptbox}

\subsection{Part (b): Efficient Histogram-Based Update (Huang's Algorithm)}

\begin{conceptbox}[Key Insight]
For 8-bit images (values 0-255), maintain a \textbf{running histogram} of the window:
\begin{itemize}
    \item \textbf{256 bins} --- one for each possible intensity value
    \item When window slides: \textbf{update} histogram (don't rebuild)
    \item Find median by \textbf{scanning} histogram (not sorting)
\end{itemize}
\end{conceptbox}

\subsubsection{Algorithm Steps}

\begin{enumerate}
    \item \textbf{Initialize:} Build histogram $H[0..255]$ for first window

    \item \textbf{Track median:} Keep count of values $\leq$ current median

    \item \textbf{When sliding right:}
    \begin{itemize}
        \item For each value $v$ in \textbf{leaving column}: $H[v] \leftarrow H[v] - 1$
        \item For each value $v$ in \textbf{entering column}: $H[v] \leftarrow H[v] + 1$
    \end{itemize}

    \item \textbf{Adjust median:}
    \begin{itemize}
        \item If removed values were $\leq$ old median: decrease count
        \item If added values were $\leq$ old median: increase count
        \item Scan histogram to find new median position
    \end{itemize}
\end{enumerate}

\section{Problem 3.24: Proving Laplacian is Isotropic}

\begin{questionbox}
\textbf{Problem 3.24:} Show that the Laplacian $\nabla^2 f = \dfrac{\partial^2 f}{\partial x^2} + \dfrac{\partial^2 f}{\partial y^2}$ is isotropic (rotation invariant).
\end{questionbox}

\subsection{Understanding Isotropy (Section 3.6.2)}

\begin{conceptbox}[What is Isotropy?]
An operator is \textbf{isotropic} (or \textbf{rotation invariant}) if it produces the \textbf{same response} regardless of the edge orientation in the image.

\textbf{Why it matters for edge detection:}
\begin{itemize}
    \item An edge at $0°$, $45°$, $90°$, or any angle should be detected equally well
    \item Anisotropic operators have ``blind spots'' for certain orientations
    \item Isotropy ensures uniform sensitivity in all directions
\end{itemize}
\end{conceptbox}

\subsection{Setting Up the Coordinate Rotation}

To prove isotropy, we show that $\nabla^2 f$ is unchanged when we \textbf{rotate} the coordinate system by angle $\theta$.

\begin{conceptbox}[Coordinate Rotation Transformation]
Rotate coordinates $(x,y)$ to new coordinates $(x', y')$:
\[
\begin{pmatrix} x \\ y \end{pmatrix} =
\begin{pmatrix} \cos\theta & -\sin\theta \\ \sin\theta & \cos\theta \end{pmatrix}
\begin{pmatrix} x' \\ y' \end{pmatrix}
\]

Explicitly:
\begin{align*}
x &= x'\cos\theta - y'\sin\theta \\
y &= x'\sin\theta + y'\cos\theta
\end{align*}
\end{conceptbox}

\begin{center}
\begin{tikzpicture}[scale=1.2]
% Original axes
\draw[->,thick] (-0.5,0) -- (3,0) node[right] {$x$};
\draw[->,thick] (0,-0.5) -- (0,2.5) node[above] {$y$};

% Rotated axes
\draw[->,thick,blue] (-0.5*0.866,-0.5*0.5) -- (2.6*0.866,2.6*0.5) node[right] {$x'$};
\draw[->,thick,red] (0.5*0.5,-0.5*0.866) -- (-2*0.5,2*0.866) node[above] {$y'$};

% Angle arc
\draw[thick] (0.8,0) arc (0:30:0.8);
\node at (1.1,0.25) {$\theta$};

% Point
\fill (1.5,1.5) circle (2pt);
\node at (1.75,1.7) {$P$};
\end{tikzpicture}
\end{center}

\subsection{Computing Partial Derivatives Under Rotation}

Using the chain rule, express derivatives in the new coordinates:

\textbf{First derivatives:}
\begin{align*}
\frac{\partial f}{\partial x'} &= \frac{\partial f}{\partial x} \cdot \frac{\partial x}{\partial x'} + \frac{\partial f}{\partial y} \cdot \frac{\partial y}{\partial x'} = f_x \cos\theta + f_y \sin\theta \\[0.5em]
\frac{\partial f}{\partial y'} &= \frac{\partial f}{\partial x} \cdot \frac{\partial x}{\partial y'} + \frac{\partial f}{\partial y} \cdot \frac{\partial y}{\partial y'} = -f_x \sin\theta + f_y \cos\theta
\end{align*}

\textbf{Second derivatives} (applying chain rule again):

\begin{align*}
\frac{\partial^2 f}{\partial x'^2} &= \frac{\partial^2 f}{\partial x^2}\cos^2\theta + 2\frac{\partial^2 f}{\partial x \partial y}\sin\theta\cos\theta + \frac{\partial^2 f}{\partial y^2}\sin^2\theta \\[0.8em]
\frac{\partial^2 f}{\partial y'^2} &= \frac{\partial^2 f}{\partial x^2}\sin^2\theta - 2\frac{\partial^2 f}{\partial x \partial y}\sin\theta\cos\theta + \frac{\partial^2 f}{\partial y^2}\cos^2\theta
\end{align*}

\subsection{The Key Step: Adding the Second Derivatives}

\begin{align*}
\nabla'^2 f &= \frac{\partial^2 f}{\partial x'^2} + \frac{\partial^2 f}{\partial y'^2} \\[0.8em]
&= \frac{\partial^2 f}{\partial x^2}\underbrace{(\cos^2\theta + \sin^2\theta)}_{= 1} + \frac{\partial^2 f}{\partial y^2}\underbrace{(\sin^2\theta + \cos^2\theta)}_{= 1} \\[0.5em]
&\quad + 2\frac{\partial^2 f}{\partial x \partial y}(\sin\theta\cos\theta - \sin\theta\cos\theta) \\[0.8em]
&= \frac{\partial^2 f}{\partial x^2} + \frac{\partial^2 f}{\partial y^2} = \nabla^2 f \quad \checkmark
\end{align*}

\begin{keypoint}
The \textbf{magic} happens because:
\begin{enumerate}
    \item $\sin^2\theta + \cos^2\theta = 1$ (Pythagorean identity)
    \item The cross term $f_{xy}$ cancels out: $+2f_{xy}\sin\theta\cos\theta - 2f_{xy}\sin\theta\cos\theta = 0$
\end{enumerate}

\textbf{Result:} $\nabla'^2 f = \nabla^2 f$ for ANY rotation angle $\theta$!
\end{keypoint}

\section{Problem 3.28: Laplacian $\approx$ Unsharp Masking}
%══════════════════════════════════════════════════════════════════════════════

\begin{questionbox}
\textbf{Problem 3.28:} Show that subtracting the Laplacian from an image is proportional to unsharp masking (Section 3.6.3).
\end{questionbox}

\subsection{Understanding Unsharp Masking (Section 3.6.3)}

\begin{conceptbox}[What is Unsharp Masking?]
A classic image sharpening technique that works by:
\begin{enumerate}
    \item Create a \textbf{blurred} (``unsharp'') version of the image
    \item \textbf{Subtract} the blur from the original to get the ``mask'' (edges/details)
    \item \textbf{Add} the mask back to enhance edges
\end{enumerate}

\textbf{Name origin:} The ``unsharp'' mask is created by blurring; adding it back makes the image sharper!
\end{conceptbox}

\subsection{Unsharp Masking Process Flow}

\begin{center}

\draw[arrow] (blur) -- (blurred);

\textbf{Mathematical formulation:}
\[
\boxed{g(x,y) = f(x,y) + k \cdot \underbrace{[f(x,y) - \bar{f}(x,y)]}_{\text{unsharp mask}}}
\]

where:
\begin{itemize}
    \item $f$ = original image
    \item $\bar{f}$ = blurred image (local average)
    \item $k$ = sharpening strength (typically $k > 0$)
    \item $g$ = sharpened output
\end{itemize}

\subsection{Understanding Laplacian Sharpening}

\begin{conceptbox}[Laplacian as Edge Detector]
The Laplacian detects edges by computing the \textbf{second derivative}:
\[
\nabla^2 f = \frac{\partial^2 f}{\partial x^2} + \frac{\partial^2 f}{\partial y^2}
\]

\textbf{Properties:}
\begin{itemize}
    \item \textbf{Positive} where intensity increases rapidly (one side of edge)
    \item \textbf{Negative} where intensity decreases rapidly (other side)
    \item \textbf{Zero} in flat regions
    \item Responds to edges regardless of direction (isotropic)
\end{itemize}
\end{conceptbox}

\textbf{Laplacian sharpening formula:}
\[
\boxed{g(x,y) = f(x,y) - c \cdot \nabla^2 f(x,y)}
\]

where $c > 0$ is a scaling constant. We \textbf{subtract} because the Laplacian gives negative values at edges we want to enhance.

\subsection{Connecting the Two Methods}

\subsubsection{Discrete Laplacian Approximation}

The continuous second derivative is approximated using differences:
\[
\frac{\partial^2 f}{\partial x^2} \approx f(x+1,y) - 2f(x,y) + f(x-1,y)
\]
\[
\frac{\partial^2 f}{\partial y^2} \approx f(x,y+1) - 2f(x,y) + f(x,y-1)
\]

\textbf{Adding them:}
\begin{align*}
\nabla^2 f &\approx f(x+1,y) + f(x-1,y) + f(x,y+1) + f(x,y-1) - 4f(x,y) \\[0.5em]
&= \underbrace{[f(x+1,y) + f(x-1,y) + f(x,y+1) + f(x,y-1)]}_{\text{sum of 4 neighbors}} - 4f(x,y)
\end{align*}

\subsubsection{Relating to Local Average}

The local average of 4 neighbors:
\[
\bar{f}(x,y) = \frac{1}{4}[f(x+1,y) + f(x-1,y) + f(x,y+1) + f(x,y-1)]
\]

Therefore:
\[
\nabla^2 f = 4\bar{f} - 4f = -4(f - \bar{f})
\]

\begin{keypoint}
\textbf{The key relationship:}
\[
\boxed{\nabla^2 f \approx -4(f - \bar{f})}
\]

The Laplacian is proportional to the \textbf{negative} of the unsharp mask!
\end{keypoint}

\subsection{Final Proof: Equivalence}

Starting with Laplacian sharpening:
\begin{align*}
g &= f - c \cdot \nabla^2 f \\[0.5em]
&= f - c \cdot (-4)(f - \bar{f}) \\[0.5em]
&= f + 4c(f - \bar{f})
\end{align*}

Setting $k = 4c$:
\[
\boxed{g = f + k(f - \bar{f})}
\]

This is exactly the unsharp masking formula!

\subsection{Practical Implications}

\begin{keypoint}
\textbf{Both methods achieve the same goal:}
\begin{enumerate}
    \item Extract high-frequency detail (edges) from the image
    \item Add this detail back to enhance sharpness
\end{enumerate}

\textbf{Relationship between parameters:}
\begin{itemize}
    \item Unsharp masking: sharpening strength $k$
    \item Laplacian: scaling constant $c = k/4$
\end{itemize}

\textbf{When to use which:}
\begin{itemize}
    \item \textbf{Unsharp masking:} More intuitive parameters, variable blur kernel size
    \item \textbf{Laplacian:} Single convolution operation, computationally simpler
\end{itemize}
\end{keypoint}

\begin{example}
\textbf{Numerical example showing equivalence:}

Given a $3 \times 3$ region with center pixel value 120 and its 4 neighbors:

\begin{center}
\begin{tikzpicture}[scale=1.1]
% Grid with proper spacing
\draw[thick] (0,0) grid[step=1.2] (3.6,3.6);

% Values - corners empty (not used in 4-neighbor Laplacian)
\node at (0.6,3) {\small ---};
\node at (1.8,3) {\small 100};
\node at (3,3) {\small ---};

\node at (0.6,1.8) {\small 80};
\node[red,font=\bfseries] at (1.8,1.8) {120};
\node at (3,1.8) {\small 90};

\node at (0.6,0.6) {\small ---};
\node at (1.8,0.6) {\small 110};
\node at (3,0.6) {\small ---};

% Label
\node at (1.8,-0.5) {\small Center pixel = \textbf{120}};
\end{tikzpicture}
\end{center}

\textbf{Step-by-step calculations:}

\begin{enumerate}
    \item \textbf{Local average} (4 neighbors):
    \[
    \bar{f} = \frac{100 + 80 + 90 + 110}{4} = \frac{380}{4} = 95
    \]

    \item \textbf{Unsharp mask} $(f - \bar{f})$:
    \[
    f - \bar{f} = 120 - 95 = 25
    \]

    \item \textbf{Discrete Laplacian}:
    \[
    \nabla^2 f = (100 + 80 + 90 + 110) - 4 \times 120 = 380 - 480 = -100
    \]

    \item \textbf{Verify the relationship}:
    \[
    -4(f - \bar{f}) = -4 \times 25 = -100 = \nabla^2 f \quad \checkmark
    \]
\end{enumerate}

\textbf{Conclusion:} The Laplacian equals $-4$ times the unsharp mask, confirming the equivalence!
\end{example}
